(

acquisition_channel VARCHAR(50) NOT NULL,

(

restaurant_category VARCHAR(100) NOT NULL,

session_id VARCHAR(50) NOT NULL, -- may be incomplete; checked later
restaurant_id INT NULL,
product_price DECIMAL(10, 2),
quantity INT,
PRIMARY KEY (event_id)
);

(
order_id INT NOT NULL,
user_id INT NOT NULL,

/* ============================================================================
STEP E.2 — SESSION RECONSTRUCTION (30-MINUTE INACTIVITY RULE)
Purpose:
- Rebuild session identifiers using event_datetime.
- Rule:
* First event for a user starts a new session.
* Any gap > 30 minutes from the previous event starts a new session.
- Output:
* events.session_id_generated = _
e.g. 100000_1, 100000_2, ...
- Non-destructive:
* Original session_id column remains as-is.
============================================================================ */
----------------------------------------------------------
-- E.2.1) Add column to store generated session IDs
----------------------------------------------------------
ALTER TABLE events
ADD COLUMN session_id_generated VARCHAR(64) NULL;
----------------------------------------------------------
-- E.2.2) Drop any previous temporary table (safety)
----------------------------------------------------------
DROP TEMPORARY TABLE IF EXISTS tmp_event_sessions;
----------------------------------------------------------
-- E.2.3) Build temporary table with session logic
-- - prev_event_dt: previous event time for the same user
-- - is_new_session: 1 if first event or gap > 30 mins
-- - session_no: cumulative sum of is_new_session per user
----------------------------------------------------------
CREATE TEMPORARY TABLE tmp_event_sessions AS
SELECT
event_id,
user_id,
event_datetime,
prev_event_dt,
CASE
WHEN prev_event_dt IS NULL THEN 1
WHEN TIMESTAMPDIFF(MINUTE, prev_event_dt, event_datetime) > 30 THEN 1
ELSE 0
END AS is_new_session,
SUM(
CASE
WHEN prev_event_dt IS NULL THEN 1
WHEN TIMESTAMPDIFF(MINUTE, prev_event_dt, event_datetime) > 30 THEN 1
ELSE 0
END
) OVER (
PARTITION BY user_id
ORDER BY event_datetime, event_id
ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
) AS session_no
FROM (
SELECT
event_id,
user_id,
event_datetime,
LAG(event_datetime) OVER (
PARTITION BY user_id
ORDER BY event_datetime, event_id
) AS prev_event_dt
FROM events
) AS sub;
----------------------------------------------------------
-- E.2.4) Index the temporary table on event_id for fast join
----------------------------------------------------------
ALTER TABLE tmp_event_sessions
ADD INDEX idx_tmp_event_sessions_event_id (event_id);
----------------------------------------------------------
-- E.2.5) Optional preview: spot-check a few computed sessions
----------------------------------------------------------
SELECT
event_id,
user_id,
event_datetime,
prev_event_dt,
is_new_session,
session_no
FROM tmp_event_sessions
ORDER BY user_id, event_datetime, event_id
LIMIT 20;
----------------------------------------------------------
-- E.2.6) Update events with generated session IDs
----------------------------------------------------------
UPDATE events AS e
JOIN tmp_event_sessions AS t
USING (event_id)
SET e.session_id_generated = CONCAT(t.user_id, '_', t.session_no);

-- E) Median events per session (MySQL 8 window-based approach)
SELECT 'events_per_session_median' AS metric,
AVG(events_per_session) AS value
FROM (
SELECT
events_per_session,
ROW_NUMBER() OVER (ORDER BY events_per_session) AS rn,
COUNT(*) OVER () AS total_rows
FROM (
SELECT COUNT(*) AS events_per_session
FROM events
GROUP BY session_id_generated
) AS x
) AS y
WHERE rn IN (
FLOOR((total_rows + 1) / 2),
CEIL((total_rows + 1) / 2)
);

-- first payment-like event from events (payment_success or order_complete)
e.t_payment,
-- earliest successful order time (success = 1)
o.first_order_time_from_orders,
-- canonical payment timestamp: prefer event payment, else order time
COALESCE(e.t_payment, o.first_order_time_from_orders) AS t_first_payment_canonical
FROM users AS u
LEFT JOIN (
-- aggregate events per user to get first timestamp for each event type
SELECT
user_id,
MIN(CASE WHEN event_type = 'browse_restaurant' THEN event_datetime END) AS t_browse,
MIN(CASE WHEN event_type = 'view_menu' THEN event_datetime END) AS t_view,
MIN(CASE WHEN event_type = 'add_to_cart' THEN event_datetime END) AS t_addcart,
MIN(CASE WHEN event_type = 'start_checkout' THEN event_datetime END) AS t_checkout,
MIN(CASE WHEN event_type IN ('payment_success', 'order_complete') THEN event_datetime END) AS t_payment
FROM events
GROUP BY user_id
) AS e
ON u.user_id = e.user_id
LEFT JOIN (
-- earliest successful order per user
SELECT
user_id,
MIN(order_datetime) AS first_order_time_from_orders
FROM orders
WHERE success = 1
GROUP BY user_id
) AS o
ON u.user_id = o.user_id
ON DUPLICATE KEY UPDATE
t_signup = VALUES(t_signup),
t_browse = VALUES(t_browse),
t_view = VALUES(t_view),
t_addcart = VALUES(t_addcart),
t_checkout = VALUES(t_checkout),
t_payment = VALUES(t_payment),
first_order_time_from_orders = VALUES(first_order_time_from_orders),
t_first_payment_canonical = VALUES(t_first_payment_canonical);

-- F.4.2) Insert weekly cohort retention rows
----------------------------------------------------------
INSERT INTO cohort_weekly_retention (
cohort_week,
cohort_start_date,
cohort_size,
week_0_active,
week_1_active,
week_2_active,
week_3_active,
week_4_active,

week_8_active,

week_3_retention_pct,
week_4_retention_pct,
week_5_retention_pct,
week_6_retention_pct,
week_7_retention_pct,
week_8_retention_pct
)
SELECT
uc.cohort_week,
uc.cohort_start_date,
uc.cohort_size,
uc.cohort_size AS week_0_active,
COALESCE(MAX(CASE WHEN a.week_index = 1 THEN a.users_active END), 0) AS week_1_active,
COALESCE(MAX(CASE WHEN a.week_index = 2 THEN a.users_active END), 0) AS week_2_active,
COALESCE(MAX(CASE WHEN a.week_index = 3 THEN a.users_active END), 0) AS week_3_active,
COALESCE(MAX(CASE WHEN a.week_index = 4 THEN a.users_active END), 0) AS week_4_active,
COALESCE(MAX(CASE WHEN a.week_index = 5 THEN a.users_active END), 0) AS week_5_active,
COALESCE(MAX(CASE WHEN a.week_index = 6 THEN a.users_active END), 0) AS week_6_active,
COALESCE(MAX(CASE WHEN a.week_index = 7 THEN a.users_active END), 0) AS week_7_active,
COALESCE(MAX(CASE WHEN a.week_index = 8 THEN a.users_active END), 0) AS week_8_active,
-- Retention percentages
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 1 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_1_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 2 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_2_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 3 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_3_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 4 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_4_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 5 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_5_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 6 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_6_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 7 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_7_retention_pct,
CASE
WHEN uc.cohort_size = 0 THEN NULL
ELSE ROUND(100.0 * COALESCE(MAX(CASE WHEN a.week_index = 8 THEN a.users_active END), 0) / uc.cohort_size, 2)
END AS week_8_retention_pct
FROM
-- User cohorts: assign each user a cohort_start_date (Monday) and cohort_week string
(
SELECT
cohort_week,
cohort_start_date,
COUNT(*) AS cohort_size
FROM (
SELECT
u.user_id,
DATE(COALESCE(u.signup_datetime, u.signup_date)) AS signup_date_only,
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
) AS cohort_start_date,
CONCAT(
YEAR(
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
)
),
'-W',
LPAD(
WEEK(
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
),
3
),
2,
'0'
)
) AS cohort_week
FROM users AS u
) AS x
GROUP BY cohort_week, cohort_start_date
) AS uc
LEFT JOIN
(
-- Activity per cohort and week_index (1..8), distinct users active
SELECT
ac.cohort_week,
ac.cohort_start_date,
ac.week_index,
COUNT(DISTINCT ac.user_id) AS users_active
FROM (
SELECT
u.user_id,
-- Cohort info from signup
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
) AS cohort_start_date,
CONCAT(
YEAR(
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
)
),
'-W',
LPAD(
WEEK(
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
),
3
),
2,
'0'
)
) AS cohort_week,
e.user_id AS event_user_id,
DATE(e.event_datetime) AS event_date,
FLOOR(
DATEDIFF(DATE(e.event_datetime),
DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
)
) / 7
) AS week_index
FROM users AS u
JOIN events AS e
ON e.user_id = u.user_id
WHERE DATE(e.event_datetime) >= DATE_SUB(
DATE(COALESCE(u.signup_datetime, u.signup_date)),
INTERVAL WEEKDAY(DATE(COALESCE(u.signup_datetime, u.signup_date))) DAY
)
) AS ac
WHERE ac.week_index BETWEEN 1 AND 8
GROUP BY ac.cohort_week, ac.cohort_start_date, ac.week_index
) AS a
ON a.cohort_week = uc.cohort_week
AND a.cohort_start_date = uc.cohort_start_date
GROUP BY
uc.cohort_week,
uc.cohort_start_date,
uc.cohort_size;

-- 1) user_segments helper table
-- One row per user with canonical stage timestamps and segment attributes.

CREATE TABLE user_segments AS
SELECT
u.user_id,
-- signup datetime: prefer user_first_stages.t_signup, then users.signup_datetime, then users.signup_date
COALESCE(ufs.t_signup, u.signup_datetime, u.signup_date) AS signup_datetime,
-- canonical payment time
COALESCE(ufs.t_first_payment_canonical, u.first_order_datetime) AS t_first_payment_canonical,
-- funnel stage timestamps
ufs.t_browse,
ufs.t_view,
ufs.t_addcart,
ufs.t_checkout,
ufs.t_payment,
-- normalized segment attributes
COALESCE(u.device, 'UNKNOWN') AS device,
COALESCE(u.acquisition_channel, 'UNKNOWN') AS acquisition_channel,
COALESCE(u.gender, 'unknown') AS gender,
COALESCE(u.city, 'unknown') AS city,
-- age_group buckets
CASE
WHEN u.age BETWEEN 18 AND 24 THEN '18-24'
WHEN u.age BETWEEN 25 AND 34 THEN '25-34'
WHEN u.age BETWEEN 35 AND 44 THEN '35-44'
WHEN u.age >= 45 THEN '45+'
WHEN u.age IS NULL THEN 'unknown'
ELSE 'unknown'
END AS age_group
FROM users AS u
LEFT JOIN user_first_stages AS ufs
ON u.user_id = ufs.user_id;

/* ============================================================================
STEP A — DAILY ACTIVITY METRICS (daily_activity_metrics)
Purpose:
- Create a per-day summary table combining:
* signups
* engagement (events/sessions)
* orders & revenue
* key conversion % and churn_7d_rate
============================================================================ */
-- Drop derived table if re-running
DROP TABLE IF EXISTS daily_activity_metrics;
CREATE TABLE daily_activity_metrics AS
WITH
-- A.1) All distinct dates present in events, orders, or signup dates
all_dates AS (
SELECT DISTINCT DATE(event_datetime) AS dt
FROM events
WHERE event_datetime IS NOT NULL
UNION
SELECT DISTINCT DATE(order_datetime) AS dt
FROM orders
WHERE order_datetime IS NOT NULL
UNION
SELECT DISTINCT DATE(COALESCE(signup_datetime, signup_date)) AS dt
FROM users
WHERE COALESCE(signup_datetime, signup_date) IS NOT NULL
),
-- A.2) Event-level aggregates per date
events_agg AS (
SELECT
DATE(e.event_datetime) AS dt,
COUNT(*) AS total_events,
COUNT(DISTINCT e.user_id) AS dau,
COUNT(DISTINCT e.session_id_generated) AS sessions,
-- users who performed a "browse" style action
COUNT(DISTINCT CASE
WHEN LOWER(e.event_type) LIKE '%browse%' THEN e.user_id
END) AS users_browsed,
-- users who performed a "view menu" style action
COUNT(DISTINCT CASE
WHEN LOWER(e.event_type) LIKE '%view%' OR LOWER(e.event_type) LIKE '%menu%'
THEN e.user_id
END) AS users_viewed,
-- users who fired a payment-like event
COUNT(DISTINCT CASE
WHEN LOWER(e.event_type) LIKE '%pay%'
OR LOWER(e.event_type) LIKE '%paid%'
OR LOWER(e.event_type) LIKE '%payment%'
OR LOWER(e.event_type) LIKE '%order_complete%'
OR LOWER(e.event_type) LIKE '%charge%'
THEN e.user_id
END) AS users_payment_event
FROM events AS e
WHERE e.event_datetime IS NOT NULL
GROUP BY DATE(e.event_datetime)
),
-- A.3) Order-level aggregates per date (successful orders only)
orders_agg AS (
SELECT
DATE(o.order_datetime) AS dt,
COUNT(*) AS total_orders,
SUM(o.order_value) AS revenue,
AVG(o.order_value) AS avg_order_value,
COUNT(DISTINCT o.user_id) AS buyers
FROM orders AS o
WHERE o.order_datetime IS NOT NULL
AND COALESCE(o.success, 0) IN (1, '1')
GROUP BY DATE(o.order_datetime)
),
-- A.4) Checkout users per date: users whose first checkout timestamp falls on this date
checkout_agg AS (
SELECT
DATE(t_checkout) AS dt,
COUNT(DISTINCT user_id) AS checkout_users
FROM user_first_stages
WHERE t_checkout IS NOT NULL
GROUP BY DATE(t_checkout)
),
-- A.5) Overlap of payment-event users and successful order buyers on same date
common_payment_order AS (
SELECT
DATE(e.event_datetime) AS dt,
COUNT(DISTINCT e.user_id) AS common_users
FROM events AS e
JOIN orders AS o
ON o.user_id = e.user_id
AND DATE(o.order_datetime) = DATE(e.event_datetime)
WHERE e.event_datetime IS NOT NULL
AND o.order_datetime IS NOT NULL
AND (
LOWER(e.event_type) LIKE '%pay%'
OR LOWER(e.event_type) LIKE '%payment%'
OR LOWER(e.event_type) LIKE '%paid%'
OR LOWER(e.event_type) LIKE '%order_complete%'
OR LOWER(e.event_type) LIKE '%charge%'
)
AND COALESCE(o.success, 0) IN (1, '1')
GROUP BY DATE(e.event_datetime)
)
SELECT
d.dt AS `date`,
-- New users who signed up on this date
COALESCE((
SELECT COUNT(*)
FROM users AS u
WHERE DATE(COALESCE(u.signup_datetime, u.signup_date)) = d.dt
), 0) AS new_users,
-- Engagement from events_agg
COALESCE(ea.dau, 0) AS dau,
COALESCE(ea.total_events, 0) AS total_events,
COALESCE(ea.sessions, 0) AS sessions,
-- Orders & revenue from orders_agg
COALESCE(oa.total_orders, 0) AS total_orders,
COALESCE(oa.revenue, 0.0) AS revenue,
CASE
WHEN COALESCE(oa.total_orders, 0) = 0 THEN NULL
ELSE ROUND(oa.revenue / NULLIF(oa.total_orders, 0), 2)
END AS avg_order_value,
/* payment_conversion_pct:
Numerator:
distinct users who either:
- fired a payment-like event that day, OR
- placed a successful order that day,
minus overlap users who did both (from common_payment_order).
Denominator:
users whose first checkout (t_checkout) fell on that date.
*/
CASE
WHEN COALESCE(ca.checkout_users, 0) = 0 THEN NULL
ELSE ROUND(
100.0 * (
COALESCE(ea.users_payment_event, 0)
+ COALESCE(oa.buyers, 0)
- COALESCE(cpo.common_users, 0)
) / NULLIF(ca.checkout_users, 0),
2
)
END AS payment_conversion_pct,
/* browse_conversion_pct:
users_viewed / users_browsed (from events_agg)
*/
CASE
WHEN COALESCE(ea.users_browsed, 0) = 0 THEN NULL
ELSE ROUND(
100.0 * COALESCE(ea.users_viewed, 0)
/ NULLIF(ea.users_browsed, 0),
2
)
END AS browse_conversion_pct,
/* churn_7d_rate:
For users who signed up on date d.dt:
churn_7d_rate = % of those NOT active on date d.dt + 7 days.
If no signups that day → NULL.
*/
CASE
WHEN (
SELECT COUNT(*)
FROM users AS u
WHERE DATE(COALESCE(u.signup_datetime, u.signup_date)) = d.dt
) = 0
THEN NULL
ELSE ROUND(
100.0 * (
(
SELECT COUNT(*)
FROM users AS u
WHERE DATE(COALESCE(u.signup_datetime, u.signup_date)) = d.dt
)
-
(
SELECT COUNT(DISTINCT u2.user_id)
FROM users AS u2
JOIN events AS e2
ON e2.user_id = u2.user_id
WHERE DATE(COALESCE(u2.signup_datetime, u2.signup_date)) = d.dt
AND DATE(e2.event_datetime) = DATE_ADD(d.dt, INTERVAL 7 DAY)
)
) / NULLIF((
SELECT COUNT(*)
FROM users AS u
WHERE DATE(COALESCE(u.signup_datetime, u.signup_date)) = d.dt
), 0),
2
)
END AS churn_7d_rate,
-- Free-text annotation field (kept NULL by default)
NULL AS notes
FROM all_dates AS d
LEFT JOIN events_agg AS ea ON ea.dt = d.dt
LEFT JOIN orders_agg AS oa ON oa.dt = d.dt
LEFT JOIN checkout_agg AS ca ON ca.dt = d.dt
LEFT JOIN common_payment_order AS cpo ON cpo.dt = d.dt
ORDER BY d.dt;
-- Indexes to support downstream analysis
ALTER TABLE daily_activity_metrics
ADD INDEX idx_dam_date (date),
ADD INDEX idx_dam_dau (dau);

/* ============================================================================
STEP B — TRAILING 7-DAY BASELINES (daily_baselines)
Purpose:
- For each date, compute trailing 7-day baselines for key metrics.
- Baseline window: D-7 .. D-1 (ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING).
============================================================================ */
DROP TABLE IF EXISTS daily_baselines;
CREATE TABLE daily_baselines AS
SELECT
dam.date AS `date`,
-- new_users
dam.new_users AS new_users_actual,
ROUND(
AVG(dam.new_users) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS new_users_baseline,
COUNT(dam.new_users) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING) AS new_users_baseline_count,
-- dau
dam.dau AS dau_actual,
ROUND(
AVG(dam.dau) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS dau_baseline,
COUNT(dam.dau) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING) AS dau_baseline_count,
-- total_events
dam.total_events AS total_events_actual,
ROUND(
AVG(dam.total_events) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS total_events_baseline,
COUNT(dam.total_events) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)
AS total_events_baseline_count,
-- sessions
dam.sessions AS sessions_actual,
ROUND(
AVG(dam.sessions) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS sessions_baseline,
COUNT(dam.sessions) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING) AS sessions_baseline_count,
-- total_orders
dam.total_orders AS total_orders_actual,
ROUND(
AVG(dam.total_orders) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS total_orders_baseline,
COUNT(dam.total_orders) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)
AS total_orders_baseline_count,
-- revenue
dam.revenue AS revenue_actual,
ROUND(
AVG(dam.revenue) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2
) AS revenue_baseline,
COUNT(dam.revenue) OVER (ORDER BY dam.date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING) AS revenue_baseline_count,
-- avg_order_value
dam.avg_order_value AS avg_order_value_actual,

ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING),
2

STEP C — DAILY ANOMALY FLAGS (daily_anomaly_flags)

-- 6) Final table: segment_daily_metrics (per date, per segment)
-------------------------------------------------------------------------------
DROP TABLE IF EXISTS segment_daily_metrics;
CREATE TABLE segment_daily_metrics AS
SELECT
d.dt AS date,
d.segment_type,
d.segment_value,
COALESCE(d.dau, 0) AS dau,
COALESCE(o.total_orders, 0) AS total_orders,
COALESCE(o.revenue, 0.0) AS revenue,
-- payment_users = payment_event_users + buyers - overlap_users
COALESCE(d.payment_event_users, 0)
+ COALESCE(o.buyers, 0)
- COALESCE(sp.overlap_users, 0) AS payment_users,
COALESCE(c.checkout_users, 0) AS checkout_users,
-- payment conversion in PERCENT (NULL when no checkout users)
CASE
WHEN COALESCE(c.checkout_users, 0) = 0 THEN NULL
ELSE ROUND(
100.0 * (
COALESCE(d.payment_event_users, 0)
+ COALESCE(o.buyers, 0)
- COALESCE(sp.overlap_users, 0)
) / NULLIF(c.checkout_users, 0),
2
)
END AS payment_conversion_pct,

D2 — segment_daily_baselines
Purpose:
- For each date × segment_type × segment_value, compute trailing 7-day

CREATE TABLE segment_daily_baselines AS
SELECT

segment_value,
-- Actuals copied from segment_daily_metrics

-- DAU baseline

AVG(dau) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
),
2

PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
) AS dau_baseline_count,

AVG(total_orders) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
),
2
) AS total_orders_baseline,

PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
) AS total_orders_baseline_count,
-- Revenue baseline
ROUND(
AVG(revenue) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
),
2
) AS revenue_baseline,
COUNT(revenue) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
) AS revenue_baseline_count,
-- Payment conversion baseline (percent, 4 decimals for a bit more precision)
ROUND(
AVG(payment_conversion_pct) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
),
4
) AS payment_conversion_pct_baseline,
COUNT(payment_conversion_pct) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
) AS payment_conversion_pct_baseline_count,
-- Browse conversion baseline (percent)
ROUND(
AVG(browse_conversion_pct) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
),
4
) AS browse_conversion_pct_baseline,
COUNT(browse_conversion_pct) OVER (
PARTITION BY segment_type, segment_value
ORDER BY date
ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
) AS browse_conversion_pct_baseline_count
FROM segment_daily_metrics
ORDER BY segment_type, segment_value, date;

COALESCE(pct_checkout_to_payment, 1000)
) = COALESCE(pct_signup_to_browse, 1000)
THEN 'signup_to_browse'
WHEN LEAST(
COALESCE(pct_signup_to_browse, 1000),
COALESCE(pct_browse_to_view, 1000),
COALESCE(pct_view_to_addcart, 1000),
COALESCE(pct_addcart_to_checkout, 1000),
COALESCE(pct_checkout_to_payment, 1000)
) = COALESCE(pct_browse_to_view, 1000)
THEN 'browse_to_view'
WHEN LEAST(
COALESCE(pct_signup_to_browse, 1000),
COALESCE(pct_browse_to_view, 1000),
COALESCE(pct_view_to_addcart, 1000),
COALESCE(pct_addcart_to_checkout, 1000),
COALESCE(pct_checkout_to_payment, 1000)
) = COALESCE(pct_view_to_addcart, 1000)
THEN 'view_to_addcart'
WHEN LEAST(
COALESCE(pct_signup_to_browse, 1000),
COALESCE(pct_browse_to_view, 1000),
COALESCE(pct_view_to_addcart, 1000),
COALESCE(pct_addcart_to_checkout, 1000),
COALESCE(pct_checkout_to_payment, 1000)
) = COALESCE(pct_addcart_to_checkout, 1000)
THEN 'addcart_to_checkout'
ELSE 'checkout_to_payment'
END AS stage_label,
LEAST(
COALESCE(pct_signup_to_browse, 1000),
COALESCE(pct_browse_to_view, 1000),
COALESCE(pct_view_to_addcart, 1000),
COALESCE(pct_addcart_to_checkout, 1000),
COALESCE(pct_checkout_to_payment, 1000)
) AS stage_value
FROM funnel_summary
LIMIT 1
) t
WHERE stage_value < 1000;
